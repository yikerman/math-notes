% Shared preamble — included by all course main.tex files

% --- Layout ---
\usepackage[margin=1in]{geometry}

% --- Math ---
\usepackage{amsmath, amssymb, amsthm}

% --- Graphs ---
\usepackage{pgfplots}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\pgfplotsset{compat=1.18}
\usetikzlibrary{calc}

% --- Boxes ---
\usepackage{tcolorbox}
\tcbuselibrary{theorems, breakable}

% --- Misc ---
\usepackage{enumitem}
\usepackage{hyperref}
\usepackage{pifont}

% --- Theorem-like environments ---

% Definition — green box, own counter
\newtcbtheorem[number within=section]{definition}{Definition}{%
	colback=red!3, colframe=red!40,
	fonttitle=\bfseries, breakable
}{def}

% --- Example box ---
\NewTColorBox{example}{ o }{%
	colback=gray!5,
	colframe=gray!60,
	fonttitle=\bfseries,
	title={Example\IfValueTF{#1}{: #1}{}},
	breakable
}

% Theorem — blue box, uses shared counter
\newtcbtheorem[number within=section]{theorem}{Theorem}{%
	colback=blue!3, colframe=blue!40,
	fonttitle=\bfseries, breakable
}{thm}

\theoremstyle{plain}

\theoremstyle{remark}
\newtheorem*{remark}{Remark}
\newtheorem*{note}{Note}
